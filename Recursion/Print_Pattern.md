```
class Solution{
    static List<Integer> pattern(int N){
        List<Integer> list = new ArrayList<>();
        helper(N, list);
        return list;
    }
    
    static void helper(int N, List<Integer> list){
        list.add(N);
        if(N<=0)
            return;
        helper(N-5, list);
        list.add(N);
    }
}
```
